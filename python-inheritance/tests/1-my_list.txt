===========================================
Doctest for MyList - Interactive Tests
===========================================

Test 1: Basic functionality - unsorted list
--------------------------------------------
>>> MyList = __import__('1-my_list').MyList
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

Test 2: Empty list
-------------------
>>> my_list = MyList()
>>> my_list.print_sorted()
[]

Test 3: Single element
-----------------------
>>> my_list = MyList([5])
>>> my_list.print_sorted()
[5]

Test 4: Already sorted list
-----------------------------
>>> my_list = MyList([1, 2, 3, 4, 5])
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]

Test 5: Reverse sorted list
-----------------------------
>>> my_list = MyList([5, 4, 3, 2, 1])
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[5, 4, 3, 2, 1]

Test 6: List with duplicate elements
--------------------------------------
>>> my_list = MyList([3, 1, 4, 1, 5, 9, 2, 6])
>>> my_list.print_sorted()
[1, 1, 2, 3, 4, 5, 6, 9]
>>> print(my_list)
[3, 1, 4, 1, 5, 9, 2, 6]

Test 7: List with negative numbers
-----------------------------------
>>> my_list = MyList([-5, -1, 0, 3, -10, 2])
>>> my_list.print_sorted()
[-10, -5, -1, 0, 2, 3]
>>> print(my_list)
[-5, -1, 0, 3, -10, 2]

Test 8: List with all same elements
------------------------------------
>>> my_list = MyList([5, 5, 5, 5])
>>> my_list.print_sorted()
[5, 5, 5, 5]

Test 9: Verify MyList is instance of list
-------------------------------------------
>>> my_list = MyList([1, 2, 3])
>>> isinstance(my_list, list)
True

Test 10: Mixed positive and negative numbers
----------------------------------------------

Test 11: Multiple print_sorted calls
--------------------------------------
>>> my_list = MyList([3, 1, 2])
>>> my_list.print_sorted()
[1, 2, 3]
>>> my_list.print_sorted()
[1, 2, 3]
>>> print(my_list)

Test 12: Two elements
----------------------
>>> my_list = MyList([2, 1])

Test 13: After append operations
---------------------------------

>>> my_list.append(8)
>>> my_list.append(1)
